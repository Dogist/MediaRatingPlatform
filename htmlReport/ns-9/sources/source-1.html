         assert path != null;
         assert httpMethod != null;
         assert method != null;
 
         this.path = path;
         this.method = method;
         this.httpMethod = httpMethod;
         this.authRequired = authRequired;
         String regexPath = path.replaceAll("\\{(\\w+)}", "(?<$1>\\\\w+)");
         this.pattern = Pattern.compile(regexPath);
     }

     }

     }
